* Build Emacs27
#+begin_src sh
  git clone --depth 1 https://github.com/emacs-mirror/emacs.git
  cd emacs
  ./autogen.sh
  ./configure --with-imagemagick --with-modules --enable-link-time-optimization --with-xwidgets
  make -j4
  sudo make install
#+end_src

* Installation
#+begin_src sh
  git clone --recursive https://github.com/zsxh/emacs.d.git ~/.emacs.d
#+end_src

* Updating
#+begin_src sh
  cd ~/.emacs.d/
  git pull
  git submodule update --init --recursive
#+end_src
